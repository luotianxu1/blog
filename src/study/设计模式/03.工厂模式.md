- 由一个工厂对象决定创建某一种产品对象类的实例。主要用来创建同一类对象

```js
class User {
    constructor(role, pages) {
        this.role = role
        this.pages = pages
    }

    static UserFactory(role) {
        switch (role) {
            case 'admin':
                return new User(role, ['home', 'dashboard', 'user', 'setting'])
            case 'user':
                return new User(role, ['home', 'dashboard'])
            default:
                return new User(role, [])
        }
    }
}

var user = User.UserFactory('admin')
console.log(user)
```
